// The Freedom Index Formula
// freedom_index.calculate()
FREEDOM_INDEX =
  (incarceration_score × 0.50) // 50% weight — core metric
  + (news_sentiment_score × 0.30) // 30% weight — daily signal

incarceration_score = max(0, (1 - (rate / 0.007))) × 100
// score 100 when rate = 0, score 0 when rate ≥ 0.7%

// normalized from -1→1 range to 0→100
